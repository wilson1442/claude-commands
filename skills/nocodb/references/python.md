### 1. `lib/nocodb.py` — NocoDB client

```python
"""
NocoDB API Client
Generated by nocodb-crud skill
"""

import os
import time
import httpx
from typing import Any, Optional

NOCODB_URL = os.environ.get("NOCODB_URL", "")
NOCODB_TABLE_ID = os.environ.get("NOCODB_TABLE_ID", "")
NOCODB_API_KEY = os.environ.get("NOCODB_API_KEY", "")


class NocoDBError(Exception):
    def __init__(self, message: str, status: int = 500, raw: Any = None):
        super().__init__(message)
        self.status = status
        self.raw = raw


def _endpoint(path: str = "") -> str:
    return f"{NOCODB_URL}/api/v2/tables/{NOCODB_TABLE_ID}{path}"


def _headers() -> dict:
    return {"xc-token": NOCODB_API_KEY, "Content-Type": "application/json"}


def _request(method: str, url: str, **kwargs) -> Any:
    last_error = None
    for attempt in range(3):
        try:
            with httpx.Client(timeout=30) as client:
                resp = client.request(method, url, headers=_headers(), **kwargs)
            if resp.status_code == 429:
                time.sleep(2 ** attempt)
                continue
            if resp.status_code == 401:
                raise NocoDBError("Auth failed — check NOCODB_API_KEY", 401)
            if resp.status_code == 404:
                raise NocoDBError("Not found", 404, resp.text)
            if not resp.is_success:
                raise NocoDBError(f"NocoDB error {resp.status_code}: {resp.text}", resp.status_code, resp.text)
            return resp.json() if resp.text else {}
        except NocoDBError:
            raise
        except Exception as e:
            last_error = e
            if attempt < 2:
                time.sleep(2 ** attempt)
    raise last_error or Exception("NocoDB request failed after retries")


def list_records(
    limit: int = 25,
    offset: int = 0,
    where: Optional[str] = None,
    sort: Optional[str] = None,
    fields: Optional[str] = None,
    view_id: Optional[str] = None,
) -> dict:
    params = {"limit": limit, "offset": offset}
    if where: params["where"] = where
    if sort: params["sort"] = sort
    if fields: params["fields"] = fields
    if view_id: params["viewId"] = view_id
    return _request("GET", _endpoint("/records"), params=params)


def get_record(row_id: int | str) -> dict:
    return _request("GET", _endpoint(f"/records/{row_id}"))


def create_record(data: dict | list[dict]) -> Any:
    return _request("POST", _endpoint("/records"), json=data)


def update_record(data: dict | list[dict]) -> Any:
    """Data must include 'Id' field for each record."""
    return _request("PATCH", _endpoint("/records"), json=data)


def delete_record(record_id: int) -> Any:
    return _request("DELETE", _endpoint("/records"), json={"Id": record_id})


def bulk_delete(ids: list[int]) -> Any:
    return _request("DELETE", _endpoint("/records"), json=[{"Id": i} for i in ids])
```

### 2a. FastAPI routes — `routes/nocodb.py`

```python
from fastapi import APIRouter, Query, HTTPException, Request
from lib.nocodb import (
    list_records, get_record, create_record,
    update_record, delete_record, bulk_delete, NocoDBError
)
from typing import Optional

router = APIRouter(prefix="/api/nocodb", tags=["nocodb"])


@router.get("/")
async def list_handler(
    limit: int = Query(25, ge=1, le=1000),
    offset: int = Query(0, ge=0),
    where: Optional[str] = None,
    sort: Optional[str] = None,
    fields: Optional[str] = None,
):
    try:
        return list_records(limit=limit, offset=offset, where=where, sort=sort, fields=fields)
    except NocoDBError as e:
        raise HTTPException(status_code=e.status, detail=str(e))


@router.get("/{row_id}")
async def get_handler(row_id: int):
    try:
        return get_record(row_id)
    except NocoDBError as e:
        raise HTTPException(status_code=e.status, detail=str(e))


@router.post("/", status_code=201)
async def create_handler(request: Request):
    try:
        body = await request.json()
        return create_record(body)
    except NocoDBError as e:
        raise HTTPException(status_code=e.status, detail=str(e))


@router.patch("/")
async def update_handler(request: Request):
    try:
        body = await request.json()
        return update_record(body)
    except NocoDBError as e:
        raise HTTPException(status_code=e.status, detail=str(e))


@router.delete("/")
async def delete_handler(request: Request):
    try:
        body = await request.json()
        if "ids" in body:
            return bulk_delete(body["ids"])
        return delete_record(body.get("id") or body.get("Id"))
    except NocoDBError as e:
        raise HTTPException(status_code=e.status, detail=str(e))
```

Mount with: `app.include_router(nocodb_router)` in your main FastAPI app.

### 2b. Flask routes — `routes/nocodb.py`

```python
from flask import Blueprint, request, jsonify
from lib.nocodb import (
    list_records, get_record, create_record,
    update_record, delete_record, bulk_delete, NocoDBError
)

nocodb_bp = Blueprint("nocodb", __name__, url_prefix="/api/nocodb")


@nocodb_bp.route("/", methods=["GET"])
def list_handler():
    try:
        return jsonify(list_records(
            limit=request.args.get("limit", 25, type=int),
            offset=request.args.get("offset", 0, type=int),
            where=request.args.get("where"),
            sort=request.args.get("sort"),
            fields=request.args.get("fields"),
        ))
    except NocoDBError as e:
        return jsonify({"error": str(e)}), e.status


@nocodb_bp.route("/<int:row_id>", methods=["GET"])
def get_handler(row_id):
    try:
        return jsonify(get_record(row_id))
    except NocoDBError as e:
        return jsonify({"error": str(e)}), e.status


@nocodb_bp.route("/", methods=["POST"])
def create_handler():
    try:
        return jsonify(create_record(request.json)), 201
    except NocoDBError as e:
        return jsonify({"error": str(e)}), e.status


@nocodb_bp.route("/", methods=["PATCH"])
def update_handler():
    try:
        return jsonify(update_record(request.json))
    except NocoDBError as e:
        return jsonify({"error": str(e)}), e.status


@nocodb_bp.route("/", methods=["DELETE"])
def delete_handler():
    try:
        body = request.json
        if "ids" in body:
            return jsonify(bulk_delete(body["ids"]))
        return jsonify(delete_record(body.get("id") or body.get("Id")))
    except NocoDBError as e:
        return jsonify({"error": str(e)}), e.status
```

Register with: `app.register_blueprint(nocodb_bp)` in your Flask app.
